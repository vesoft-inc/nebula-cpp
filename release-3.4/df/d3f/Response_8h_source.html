286 template <typename T>
287 bool inline checkPointer(const T *lhs, const T *rhs) {
288  if (lhs == rhs) {
289  return true;
290  } else if (lhs != nullptr && rhs != nullptr) {
291  return *lhs == *rhs;
292  } else {
293  return false;
